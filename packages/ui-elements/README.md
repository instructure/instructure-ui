### Installation

```sh
yarn add @instructure/ui-elements
```

### Usage

```js
import React from 'react'
import { Avatar } from '@instructure/ui-elements'

const MyAvatar = () => {
  return (
    <Avatar name="Kyle Montgomery" />
  )
}
```
